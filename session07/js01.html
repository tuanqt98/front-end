    4.2 Cấu trúc lặp 
    a) for  (4 dạng )
             Dạng 1: for đầy đủ :
             systax: 
             for(init ; condition ; step) {
                    statement;
             }
             Dạng 2 : for khuyết 
             systax : 
             for ( ; condition ; step) {
                 statement;
             }
             systax : 
             for ( init ;  ; step) {
                 statement;
             }
             systax : 
             for (init ; condition  ; ) {
                 statement;
             }
             * ) for khuyết toàn bộ 
             systax:
             for( ; ; ){
                 statement;
             }
             *) for không thân 
             systax:
             for(init ; condition ; step ; statement);

    b) while 
             systax : 
             while (condition) {
                 statement;
             }

    c) do while 
             systax:
             do {
                 statement;
             }while(condition);


    d for each
    for (var item in object) {
        statement;
    }
